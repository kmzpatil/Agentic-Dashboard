[Test 1/1] Question: "How many videos are being uploaded, processed, and published over time?"
  Querying Agent API (timeout=180s)...
  API Response Time: 14.63s
  ✅ PASS

  [AGENT SQL]
  SELECT 
      date_trunc('month', to_date(rv."Upload_Date", 'YYYY-MM-DD'))::date AS upload_period,
      COUNT(DISTINCT rv."Video_ID") AS uploaded_videos,
      COUNT(DISTINCT ca."Asset_ID") AS processed_assets,
      COUNT(DISTINCT pp."Post_ID") AS published_posts
  FROM
      "raw_videos" AS rv
  LEFT JOIN "created_assets" AS ca ON ca."Video_ID" = rv."Video_ID"
  LEFT JOIN "published_posts" AS pp ON pp."Asset_ID" = ca."Asset_ID"
  GROUP BY 1
  ORDER BY 1
